   362тЖТ   362тЖТ   362тЖТfunction parsePosts(raw: string, platform: string): string[] {
   363тЖТ   363тЖТ   363тЖТ  const profile = PLATFORM_PROFILES[platform] || PLATFORM_PROFILES.twitter;
   364тЖТ   364тЖТ   364тЖТ  let posts: string[] = [];
   365тЖТ   365тЖТ   365тЖТ
   366тЖТ   366тЖТ   366тЖТ  // Pre-clean: strip trailing incomplete delimiter artifacts at the very end
   367тЖТ   367тЖТ   367тЖТ  if (raw.endsWith("POST---")) {
   368тЖТ   368тЖТ   368тЖТ    raw = raw.slice(0, -8).trim();
   369тЖТ   369тЖТ   369тЖТ  }
   370тЖТ   370тЖТ   370тЖТ
   371тЖТ   371тЖТ   371тЖТ  // Strategy 1: Primary delimiter (most reliable)
   372тЖТ   372тЖТ   372тЖТ  if (raw.includes("---POST---")) {
   373тЖТ   373тЖТ   373тЖТ    posts = raw.split("---POST---").map((p) => p.trim()).filter((p) => p.length > 15);
   374тЖТ   374тЖТ   374тЖТ  }
   375тЖТ   375тЖТ   375тЖТ
   376тЖТ   376тЖТ   376тЖТ  // Strategy 2: Numbered list pattern (1. 2. 3. etc.)
   377тЖТ   377тЖТ   377тЖТ  if (posts.length <= 1) {
   378тЖТ   378тЖТ   378тЖТ    const numbered = raw
   379тЖТ   379тЖТ   379тЖТ      .split(/(?:^|\n)\s*\d+[\.\)]\s+/)
   380тЖТ   380тЖТ   380тЖТ      .map((p) => p.trim())
   381тЖТ   381тЖТ   381тЖТ      .filter((p) => p.length > 15);
   382тЖТ   382тЖТ   382тЖТ    if (numbered.length > posts.length) posts = numbered;
   383тЖТ   383тЖТ   383тЖТ  }
   384тЖТ   384тЖТ   384тЖТ
   385тЖТ   385тЖТ   385тЖТ  // Strategy 3: Split by hashtag clusters (works for short AND long platforms)
   386тЖТ   386тЖТ   386тЖТ  if (posts.length <= 1) {
   387тЖТ   387тЖТ   387тЖТ    const hashSplit = raw
   388тЖТ   388тЖТ   388тЖТ      .split(/\n\s*\n(?=#)/)
   389тЖТ   389тЖТ   389тЖТ      .map((p) => p.trim())
   390тЖТ   390тЖТ   390тЖТ      .filter((p) => p.length > 20);
   391тЖТ   391тЖТ   391тЖТ    if (hashSplit.length > posts.length) posts = hashSplit;
   392тЖТ   392тЖТ   392тЖТ  }
   393тЖТ   393тЖТ   393тЖТ
   394тЖТ   394тЖТ   394тЖТ  // Strategy 4: For long platforms, split by repeated "Post #N" labels
   395тЖТ   395тЖТ   395тЖТ  if (posts.length <= 1) {
   396тЖТ   396тЖТ   396тЖТ    const labelSplit = raw
   397тЖТ   397тЖТ   397тЖТ      .split(/(?:(?:^|\n)\s*(?:Post|POST)\s*#\d+[\s:\-]*)/i)
   398тЖТ   398тЖТ   398тЖТ      .map((p) => p.trim())
   399тЖТ   399тЖТ   399тЖТ      .filter((p) => p.length > 50);
   400тЖТ   400тЖТ   400тЖТ    if (labelSplit.length > posts.length) posts = labelSplit;
   401тЖТ   401тЖТ   401тЖТ  }
   402тЖТ   402тЖТ   402тЖТ
   403тЖТ   403тЖТ   403тЖТ  // Smart split: for ANY platform, if a post is way over maxChars,
   404тЖТ   404тЖТ   404тЖТ  // it likely contains multiple posts merged. Split at natural boundaries.
   405тЖТ   405тЖТ   405тЖТ  {
   406тЖТ   406тЖТ   406тЖТ    const expanded: string[] = [];
   407тЖТ   407тЖТ   407тЖТ    for (const post of posts) {
   408тЖТ   408тЖТ   408тЖТ      if (post.length > profile.maxChars * 1.1) {
   409тЖТ   409тЖТ   409тЖТ        let chunks: string[] = [];
   410тЖТ   410тЖТ   410тЖТ
   411тЖТ   411тЖТ   411тЖТ        // Strategy A: Split at double-newline + hashtag (good for IG, YT)
   412тЖТ   412тЖТ   412тЖТ        chunks = post.split(/\n\n(?=#)/).map((c) => c.trim()).filter((c) => c.length > 15);
   413тЖТ   413тЖТ   413тЖТ
   414тЖТ   414тЖТ   414тЖТ        // Strategy B: Split at double-newline + emoji/numbered header
   415тЖТ   415тЖТ   415тЖТ        if (chunks.length <= 1) {
   416тЖТ   416тЖТ   416тЖТ          chunks = post.split(/\n\s*\n(?=[\d#\-*])/).map((c) => c.trim()).filter((c) => c.length > 15);
   417тЖТ   417тЖТ   417тЖТ        }
   418тЖТ   418тЖТ   418тЖТ
   419тЖТ   419тЖТ   419тЖТ        // Strategy C: Split at "Post #N" labels within the merged block
   420тЖТ   420тЖТ   420тЖТ        if (chunks.length <= 1) {
   421тЖТ   421тЖТ   421тЖТ          chunks = post.split(/(?:^|\n)\s*Post\s*#\d+[\s:\-]*/i).map((c) => c.trim()).filter((c) => c.length > 50);
   422тЖТ   422тЖТ   422тЖТ        }
   423тЖТ   423тЖТ   423тЖТ

   426тЖТ   426тЖТ   426тЖТ          chunks = post.split(/\n\s*\n/).map((c) => c.trim()).filter((c) => c.length > 100);
   427тЖТ   427тЖТ   427тЖТ        }

   429тЖТ   429тЖТ   429тЖТ        if (chunks.length > 1) {
   430тЖТ   430тЖТ   430тЖТ          expanded.push(...chunks);
   431тЖТ   431тЖТ   431тЖТ        } else {
   432тЖТ   432тЖТ   432тЖТ          expanded.push(post);
   433тЖТ   433тЖТ   433тЖТ        }
   434тЖТ   434тЖТ   434тЖТ      } else {
   435тЖТ   435тЖТ   435тЖТ        expanded.push(post);
   436тЖТ   436тЖТ   436тЖТ      }

   439тЖТ   439тЖТ   439тЖТ  }
   440тЖТ   440тЖТ   440тЖТ
   441тЖТ   441тЖТ   441тЖТ  // Clean each post

   443тЖТ   443тЖТ   443тЖТ    p

   447тЖТ   447тЖТ   447тЖТ      .trim()
   448тЖТ   448тЖТ   448тЖТ  );
   449тЖТ   449тЖТ   449тЖТ
   450тЖТ   450тЖТ   450тЖТ  // Filter out posts that are clearly too short or too long for the platform
   451тЖТ   451тЖТ   451тЖТ  posts = posts.filter((p) => {
   452тЖТ   452тЖТ   452тЖТ    if (p.length < 15) return false;
   453тЖТ   453тЖТ   453тЖТ    if (p.length > profile.maxChars * 1.5) return false;
   454тЖТ   454тЖТ   454тЖТ    return true;
   455тЖТ   455тЖТ   455тЖТ  });
   456тЖТ   456тЖТ   456тЖТ
   457тЖТ   457тЖТ   457тЖТ  return posts;
   458тЖТ   458тЖТ   458тЖТ}

   501тЖТ   501тЖТ   501тЖТinterface CreditInfo {
   502тЖТ   502тЖТ   502тЖТ  allowed: boolean;

   509тЖТ   509тЖТ   509тЖТasync function checkAndDeductCredits(userId: string, cost: number): Promise<CreditInfo> {
   510тЖТ   510тЖТ   510тЖТ  const user = await db.user.findUnique({ where: { id: userId } });
   511тЖТ   511тЖТ   511тЖТ  if (!user) return { allowed: true, remaining: 20, cost, plan: 'free', maxCredits: 20 };
   512тЖТ   512тЖТ   512тЖТ
   513тЖТ   513тЖТ   513тЖТ  // Enterprise = unlimited
   514тЖТ   514тЖТ   514тЖТ  if (user.plan === 'enterprise') {
   515тЖТ   515тЖТ   515тЖТ    return { allowed: true, remaining: 99999, cost, plan: 'enterprise', maxCredits: 99999 };
   516тЖТ   516тЖТ   516тЖТ  }
   517тЖТ   517тЖТ   517тЖТ
   518тЖТ   518тЖТ   518тЖТ  const maxCredits = getCreditsForPlan(user.plan);
   519тЖТ   519тЖТ   519тЖТ  const remaining = user.credits;
   520тЖТ   520тЖТ   520тЖТ
   521тЖТ   521тЖТ   521тЖТ  if (remaining < cost) {
   522тЖТ   522тЖТ   522тЖТ    return { allowed: false, remaining: 0, cost, plan: user.plan, maxCredits };
   523тЖТ   523тЖТ   523тЖТ  }
   524тЖТ   524тЖТ   524тЖТ
   525тЖТ   525тЖТ   525тЖТ  // Deduct credits
   526тЖТ   526тЖТ   526тЖТ  await db.user.update({
   527тЖТ   527тЖТ   527тЖТ    where: { id: userId },
   528тЖТ   528тЖТ   528тЖТ    data: { credits: { decrement: cost } },
   529тЖТ   529тЖТ   529тЖТ  });
   530тЖТ   530тЖТ   530тЖТ
   531тЖТ   531тЖТ   531тЖТ  return {
   532тЖТ   532тЖТ   532тЖТ    allowed: true,
   533тЖТ   533тЖТ   533тЖТ    remaining: user.credits - cost,
   534тЖТ   534тЖТ   534тЖТ    cost,
   535тЖТ   535тЖТ   535тЖТ    plan: user.plan,
   536тЖТ   536тЖТ   536тЖТ    maxCredits,
   537тЖТ   537тЖТ   537тЖТ  };
   538тЖТ   538тЖТ   538тЖТ}
