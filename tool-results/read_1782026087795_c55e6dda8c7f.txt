   393→  /* ---------------------------------------------------------------- */
   394→  /*  Enhance Prompt handler                                           */
   395→  /* ---------------------------------------------------------------- */
   396→
   397→  async function handleEnhance() {
   398→    const text = inputMode === 'topic' ? topic.trim() : repurposeText.trim();
   399→    if (!text || text.length < 3) {
   400→      toast.error('Enter something first so the AI can enhance it');
   401→      return;
   402→    }
   403→    setIsEnhancing(true);
   404→    try {
   405→      const res = await fetch('/api/generate', {
   406→        method: 'POST',
   407→        headers: { 'Content-Type': 'application/json' },
   408→        body: JSON.stringify({
   409→          topic: text,
   410→          platform: 'linkedin',
   411→          tone: 'educational',
   412→          count: 1,
   413→          mode: 'enhance',
   414→        }),
   415→      });
   416→      const data = await res.json();
   417→      if (data.success && data.enhanced) {
   418→        if (inputMode === 'topic') {
   419→          setTopic(data.enhanced);
   420→        } else {
   421→          setRepurposeText(data.enhanced);
   422→        }
   423→        toast.success('Prompt enhanced! Your topic is now more detailed and specific.');
   424→      } else {
   425→        toast.error('Could not enhance. Try again.');
   426→      }
   427→    } catch {
   428→      toast.error('Network error while enhancing.');
   429→    } finally {
   430→      setIsEnhancing(false);
   431→    }
   432→  }
   433→
   434→  /* ---------------------------------------------------------------- */
   435→  /*  Generate handler                                                 */
   436→  /* ---------------------------------------------------------------- */
   437→
   438→  async function handleGenerate() {
   439→    const mainTopic = inputMode === 'topic' ? topic.trim() : repurposeText.trim();
   440→    if (!mainTopic || mainTopic.length < 3) {
   441→      toast.error('Please enter a topic or content to repurpose (at least 3 characters)');
   442→      return;
   443→    }
   444→    if (selectedPlatforms.length === 0) {
   445→      toast.error('Select at least one platform');
   446→      return;
   447→    }
   448→
   449→    setIsGenerating(true);
   450→    setGeneratedPosts([]);
   451→
   452→    try {
   453→      // If multi-platform, generate for each selected platform
   454→      const allResults: GeneratedPost[] = [];
   455→      
   456→      for (const platform of selectedPlatforms) {
   457→        const res = await fetch('/api/generate', {
   458→          method: 'POST',
   459→          headers: { 'Content-Type': 'application/json' },
   460→          body: JSON.stringify({
   461→            topic: mainTopic,
   462→            platform,
   463→            tone: selectedTone,
   464→            count: postCount,
   465→            mode: inputMode === 'repurpose' ? 'repurpose' : 'generate',
   466→            brandVoice: brandVoice.trim() || undefined,
   467→            userId: status === 'authenticated' ? (session?.user as any)?.id : undefined,
   468→          }),
   469→        });
   470→
   471→        const data = await res.json();
   472→
   473→        if (res.status === 403) {
   474→          toast.error(data.error || 'No credits remaining. Upgrade your plan!');
   475→          setCredits(data.creditsRemaining ?? 0);
   476→          setIsGenerating(false);
   477→          return;
   478→        }
   479→
   480→        if (!res.ok || !data.success) {
   481→          toast.error(data.error || `Generation failed for ${platform}`);
   482→          continue;
   483→        }
   484→
   485→        // Update credits from response
   486→        if (data.creditsRemaining !== undefined) {
   487→          setCredits(data.creditsRemaining);
   488→        }
   489→        if (data.plan) setUserPlan(data.plan);
   490→
   491→        const posts: GeneratedPost[] = data.posts.map((content: string) => ({
   492→          id: uid(),
   493→          content,
   494→          platform: data.platform,
   495→          copied: false,
   496→        }));
   497→
   498→        allResults.push(...posts.slice(0, postCount));
   499→      }
   500→
   501→      setGeneratedPosts(allResults);
   502→      setGenerationsUsed((prev) => prev + 1);
   503→      
   504→      const platformNames = selectedPlatforms.map((id) => PLATFORMS.find((p) => p.id === id)?.name).filter(Boolean).join(', ');
   505→      toast.success(`Generated ${allResults.length} posts across ${platformNames}!`);
   506→    } catch {
   507→      toast.error('Network error. Please check your connection and try again.');
   508→    } finally {
   509→      setIsGenerating(false);
   510→    }
   511→  }

   517→  function handleCopy(postId: string, text: string) {

   519→      setGeneratedPosts((prev) => prev.map((p) => (p.id === postId ? { ...p, copied: true } : p)));

   521→      setTimeout(() => {
   522→        setGeneratedPosts((prev) => prev.map((p) => (p.id === postId ? { ...p, copied: false } : p)));
   523→      }, 2000);
   524→    });
   525→  }
