   256→async function generateWithAI(prompt: string): Promise<string> {
   257→  const systemPrompt = `You are PostPilot, the world's best social media content strategist. You have 15 years of experience helping brands go viral. You understand each platform's algorithm, culture, and content format at an expert level. Your content never feels AI-generated — it feels like it was written by a top-tier creator who genuinely lives and breathes that platform. You NEVER violate platform-specific format rules.`;
   258→
   259→  const res = await fetch("https://text.pollinations.ai/openai/chat/completions", {
   260→    method: "POST",
   261→    headers: { "Content-Type": "application/json" },
   262→    body: JSON.stringify({
   263→      model: "openai",
   264→      messages: [
   265→        { role: "system", content: systemPrompt },
   266→        { role: "user", content: prompt },
   267→      ],
   268→      temperature: 0.9,
   269→    }),
   270→    signal: AbortSignal.timeout(60000), // 60s timeout
   271→  });
   272→
   273→  if (!res.ok) {
   274→    const errText = await res.text().catch(() => "Unknown error");
   275→    throw new Error(`Pollinations API error ${res.status}: ${errText}`);
   276→  }
   277→
   278→  const data = await res.json();
   279→  const content = data.choices?.[0]?.message?.content;
   280→  if (!content || content.trim().length < 5) {
   281→    console.error("Pollinations returned empty:", JSON.stringify(data).substring(0, 500));
   282→    throw new Error("AI returned empty content. Please try again.");
   283→  }
   284→  return content;
   285→}

   460→/*  Post parser — multi-strategy with length validation                */
   461→/* ------------------------------------------------------------------ */
   462→
   463→function parsePosts(raw: string, platform: string): string[] {
   464→  const profile = PLATFORM_PROFILES[platform] || PLATFORM_PROFILES.twitter;
   465→  let posts: string[] = [];
   466→
   467→  // Pre-clean: strip trailing incomplete delimiter artifacts at the very end
   468→  if (raw.endsWith("POST---")) {
   469→    raw = raw.slice(0, -8).trim();
   470→  }
   471→
   472→  // Strategy 1: Primary delimiter (most reliable)
   473→  if (raw.includes("---POST---")) {
   474→    posts = raw.split("---POST---").map((p) => p.trim()).filter((p) => p.length > 15);
   475→  }
   476→
   477→  // Strategy 2: Numbered list pattern (1. 2. 3. etc.)
   478→  if (posts.length <= 1) {
   479→    const numbered = raw
   480→      .split(/(?:^|\n)\s*\d+[\.\)]\s+/)
   481→      .map((p) => p.trim())
   482→      .filter((p) => p.length > 15);
   483→    if (numbered.length > posts.length) posts = numbered;
   484→  }
   485→
   486→  // Strategy 3: Split by hashtag clusters (works for short AND long platforms)
   487→  if (posts.length <= 1) {
   488→    const hashSplit = raw
   489→      .split(/\n\s*\n(?=#)/)
   490→      .map((p) => p.trim())
   491→      .filter((p) => p.length > 20);
   492→    if (hashSplit.length > posts.length) posts = hashSplit;
   493→  }
   494→
   495→  // Strategy 4: For long platforms, split by repeated "Post #N" labels
   496→  if (posts.length <= 1) {
   497→    const labelSplit = raw
   498→      .split(/(?:(?:^|\n)\s*(?:Post|POST)\s*#\d+[\s:\-]*)/i)
   499→      .map((p) => p.trim())
   500→      .filter((p) => p.length > 50);
   501→    if (labelSplit.length > posts.length) posts = labelSplit;
   502→  }
   503→
   504→  // Smart split: for ANY platform, if a post is way over maxChars,
   505→  // it likely contains multiple posts merged. Split at natural boundaries.
   506→  {
   507→    const expanded: string[] = [];
   508→    for (const post of posts) {
   509→      if (post.length > profile.maxChars * 1.1) {
   510→        let chunks: string[] = [];
   511→
   512→        // Strategy A: Split at double-newline + hashtag (good for IG, YT)
   513→        chunks = post.split(/\n\n(?=#)/).map((c) => c.trim()).filter((c) => c.length > 15);
   514→
   515→        // Strategy B: Split at double-newline + emoji/numbered header
   516→        if (chunks.length <= 1) {
   517→          chunks = post.split(/\n\s*\n(?=[\d#\-*])/).map((c) => c.trim()).filter((c) => c.length > 15);
   518→        }
   519→
   520→        // Strategy C: Split at "Post #N" labels within the merged block
   521→        if (chunks.length <= 1) {
   522→          chunks = post.split(/(?:^|\n)\s*Post\s*#\d+[\s:\-]*/i).map((c) => c.trim()).filter((c) => c.length > 50);
   523→        }
   524→
   525→        // Strategy D: For very long posts, try double newline boundaries
   526→        if (chunks.length <= 1 && post.length > profile.maxChars * 1.5) {
   527→          chunks = post.split(/\n\s*\n/).map((c) => c.trim()).filter((c) => c.length > 100);
   528→        }
   529→
   530→        if (chunks.length > 1) {
   531→          expanded.push(...chunks);
   532→        } else {
   533→          expanded.push(post);
   534→        }
   535→      } else {
   536→        expanded.push(post);
   537→      }
   538→    }
   539→    posts = expanded;
   540→  }
   541→
   542→  // Clean each post
   543→  posts = posts.map((p) =>
   544→    p
   545→      .replace(/^\s*Post\s*\d+[\s:\-]*/i, "")
   546→      .replace(/^\s*[-—]{2,}\s*/gm, "")
   547→      .replace(/^\s*#{1,}\s*Post\s*\d+/i, "")
   548→      .trim()
   549→  );
   550→
   551→  // Filter out posts that are clearly too short or too long for the platform
   552→  posts = posts.filter((p) => {
   553→    if (p.length < 15) return false;
   554→    if (p.length > profile.maxChars * 1.5) return false;
   555→    return true;
   556→  });
   557→
   558→  return posts;
   559→}

   565→const RATE_LIMIT_MAX = 5;
   566→const RATE_LIMIT_WINDOW = 60_000; // 1 minute
   567→
   568→async function isRateLimited(ip: string): Promise<boolean> {
   569→  try {
   570→    const since = new Date(Date.now() - RATE_LIMIT_WINDOW);
   571→    const recentCount = await db.rateLimitLog.count({
   572→      where: { ip, createdAt: { gte: since } },
   573→    });
   574→    return recentCount >= RATE_LIMIT_MAX;
   575→  } catch {
   576→    return false; // If DB fails, allow requests
   577→  }
   578→}

   580→async function logRateLimit(ip: string): Promise<void> {
   581→  try {
   582→    await db.rateLimitLog.create({ data: { ip } }).catch(() => {});
   583→  } catch {}
   584→}

   586→/* ------------------------------------------------------------------ */
   587→/*  Credit system — cost per generation, deduct from user                */
   588→/* ------------------------------------------------------------------ */
   589→
   590→// Credit costs per action
   591→const CREDIT_COST_GENERATE = 1;   // 1 credit per generation
   592→const CREDIT_COST_ENHANCE  = 0;   // free — encourage users to write better prompts
   593→
   594→// Plan limits
   595→const PLAN_CREDITS: Record<string, number> = {
   596→  free: 20,
   597→  pro: 500,
   598→  enterprise: 99999, // effectively unlimited
   599→};
   600→
   601→function getCreditsForPlan(plan: string): number {
   602→  return PLAN_CREDITS[plan] || PLAN_CREDITS.free;
   603→}
   604→
   605→interface CreditInfo {
   606→  allowed: boolean;
   607→  remaining: number;
   608→  cost: number;
   609→  plan: string;
   610→  maxCredits: number;
   611→}
   612→
   613→async function checkAndDeductCredits(userId: string, cost: number): Promise<CreditInfo> {
   614→  try {
   615→    const user = await db.user.findUnique({ where: { id: userId } });
   616→    if (!user) return { allowed: true, remaining: 20, cost, plan: 'free', maxCredits: 20 };
   617→    if (user.plan === 'enterprise') return { allowed: true, remaining: 99999, cost, plan: 'enterprise', maxCredits: 99999 };
   618→    const maxCredits = getCreditsForPlan(user.plan);
   619→    const remaining = user.credits;
   620→    if (remaining < cost) return { allowed: false, remaining: 0, cost, plan: user.plan, maxCredits };
   621→    await db.user.update({ where: { id: userId }, data: { credits: { decrement: cost } } });
   622→    return { allowed: true, remaining: user.credits - cost, cost, plan: user.plan, maxCredits };
   623→  } catch {
   624→    return { allowed: true, remaining: 20, cost, plan: 'free', maxCredits: 20 };
   625→  }
   626→}

   629→/*  GET credits — endpoint for frontend to check balance               */
   630→/* ------------------------------------------------------------------ */
   631→
   632→export async function GET(request: NextRequest) {
   633→  try {
   634→    const userId = request.nextUrl.searchParams.get('userId');
   635→    const email = request.nextUrl.searchParams.get('email');
   636→    let user = null;
   637→    try {
   638→      if (userId) user = await db.user.findUnique({ where: { id: userId }, select: { credits: true, plan: true, createdAt: true } });
   639→      if (!user && email) user = await db.user.findUnique({ where: { email }, select: { credits: true, plan: true, createdAt: true } });
   640→    } catch {}
   641→    if (!user) return NextResponse.json({ credits: 20, plan: 'free', maxCredits: 20 });
   642→    return NextResponse.json({ credits: user.credits, plan: user.plan, maxCredits: getCreditsForPlan(user.plan) });
   643→  } catch (error) {
   644→    return NextResponse.json({ credits: 20, plan: 'free', maxCredits: 20 });
   645→  }
   646→}

   648→/* ------------------------------------------------------------------ */
   649→/*  API Route                                                          */
   650→/* ------------------------------------------------------------------ */
   651→
   652→export async function POST(request: NextRequest) {
   653→  try {
   654→    const ip = request.headers.get("x-forwarded-for")?.split(",")[0]?.trim() || "unknown";
   655→
   656→    // Rate limiting by IP (database-backed)
   657→    if (await isRateLimited(ip)) {
   658→      await logRateLimit(ip);
   659→      return NextResponse.json(
   660→        { error: "Rate limit reached. Please wait a moment before generating more posts." },
   661→        { status: 429 }
   662→      );
   663→    }
   664→
   665→    const body = await request.json();
   666→    const { topic, platform, tone, count, mode, brandVoice, userId } = body as GenerateRequest;
   667→
   668→    if (!topic || topic.trim().length < 3) {
   669→      return NextResponse.json({ error: "Please provide a topic (at least 3 characters)" }, { status: 400 });
   670→    }
   671→
   672→    // ENHANCE MODE — just return an enhanced prompt, don't generate posts
   673→    if (mode === "enhance") {
   674→      const enhancePrompt = buildEnhancePrompt(topic.trim());
   675→      const enhanced = await generateWithAI(enhancePrompt);
   676→      return NextResponse.json({
   677→        success: true,
   678→        enhanced: enhanced.trim().replace(/^["']|["']$/g, ""),
   679→      });
   680→    }
   681→
   682→    // HOOK MODE — generate viral hooks
   683→    if (mode === "hook") {
   684→      const hookPrompt = buildHookPrompt({ topic, platform: platform || "twitter", tone: tone || "casual", count: 5, mode: "hook", brandVoice, userId });
   685→      const rawContent = await generateWithAI(hookPrompt);
   686→      let hooks = rawContent.split("---POST---").map((h) => h.trim()).filter((h) => h.length > 10);
   687→      if (hooks.length <= 1) {
   688→        hooks = rawContent.split(/\n(?=[-\d]+\.\s)/).map((h) => h.trim()).filter((h) => h.length > 10);
   689→      }
   690→      if (hooks.length <= 1) {
   691→        hooks = rawContent.split(/\n{2,}/).map((h) => h.trim()).filter((h) => h.length > 10);
   692→      }
   693→      return NextResponse.json({ success: true, hooks, topic, tone: tone || "casual", generatedAt: new Date().toISOString() });
   694→    }
   695→
   696→    // THREAD MODE — generate Twitter thread
   697→    if (mode === "thread") {
   698→      const threadPrompt = buildThreadPrompt({ topic, platform: "twitter", tone: tone || "casual", count: 7, mode: "thread", brandVoice, userId });
   699→      const rawContent = await generateWithAI(threadPrompt);
   700→      
   701→      // Try multiple delimiters
   702→      let tweets = rawContent.split("---POST---").map((t) => t.trim()).filter((t) => t.length > 10);
   703→      if (tweets.length <= 1) {
   704→        tweets = rawContent.split(/\n(?=\d+\/)/).map((t) => t.trim()).filter((t) => t.length > 10);
   705→      }
   706→      if (tweets.length <= 1) {
   707→        tweets = rawContent.split(/\n{2,}/).map((t) => t.trim()).filter((t) => t.length > 10);
   708→      }
   709→      
   710→      return NextResponse.json({ success: true, tweets, topic, tone: tone || "casual", generatedAt: new Date().toISOString() });
   711→    }
   712→
   713→    // CAROUSEL MODE — generate Instagram carousel
   714→    if (mode === "carousel") {
   715→      const carouselPrompt = buildCarouselPrompt({ topic, platform: "instagram", tone: tone || "casual", count: 7, mode: "carousel", brandVoice, userId });
   716→      const rawContent = await generateWithAI(carouselPrompt);
   717→      let slides = rawContent.split("---POST---").map((s) => s.trim()).filter((s) => s.length > 10);
   718→      if (slides.length <= 1) {
   719→        slides = rawContent.split(/\n{2,}/).map((s) => s.trim()).filter((s) => s.length > 10);
   720→      }
   721→      return NextResponse.json({ success: true, slides, topic, tone: tone || "casual", generatedAt: new Date().toISOString() });
   722→    }
   723→
   724→    // HASHTAGS MODE — generate hashtag suggestions
   725→    if (mode === "hashtags") {
   726→      const hashtagPrompt = buildHashtagPrompt({ topic, platform: platform || "instagram", tone: tone || "casual", count: 15, mode: "hashtags", brandVoice, userId });
   727→      const rawContent = await generateWithAI(hashtagPrompt);
   728→      return NextResponse.json({ success: true, rawHashtags: rawContent.trim(), topic, platform: platform || "instagram", generatedAt: new Date().toISOString() });
   729→    }
   730→
   731→    // Credit check for authenticated users
   732→    if (userId) {
   733→      const creditCheck = await checkAndDeductCredits(userId, CREDIT_COST_GENERATE);
   734→      if (!creditCheck.allowed) {
   735→        return NextResponse.json(
   736→          { error: "No credits remaining. Upgrade your plan for more credits.", creditsRemaining: 0, plan: creditCheck.plan },
   737→          { status: 403 }
   738→        );
   739→      }
   740→    }
   741→
   742→    const validPlatforms = Object.keys(PLATFORM_PROFILES);
   743→    if (!platform || !validPlatforms.includes(platform)) {
   744→      return NextResponse.json({ error: `Invalid platform. Choose from: ${validPlatforms.join(", ")}` }, { status: 400 });
   745→    }
   746→
   747→    const validTones = Object.keys(TONE_INSTRUCTIONS);
   748→    if (!tone || !validTones.includes(tone)) {
   749→      return NextResponse.json({ error: `Invalid tone. Choose from: ${validTones.join(", ")}` }, { status: 400 });
   750→    }
   751→
   752→    const postCount = Math.min(Math.max(count || 3, 1), 10);
   753→    const prompt = buildPrompt({ topic, platform, tone, count: postCount, mode: mode || "generate", brandVoice, userId });
   754→
   755→    const rawContent = await generateWithAI(prompt);
   756→    const posts = parsePosts(rawContent, platform);
   757→
   758→    if (posts.length === 0) {
   759→      return NextResponse.json({ error: "Failed to parse generated content. Please try again." }, { status: 500 });
   760→    }
   761→
   762→    // Log generation to database (fire-and-forget)
   763→    db.generation.create({
   764→      data: {
   765→        userId: userId || null,
   766→        topic: topic.trim(),
   767→        platform,
   768→        tone,
   769→        postCount,
   770→        postsGenerated: posts.length,
   771→        mode: mode || "generate",
   772→        postsJson: JSON.stringify(posts),
   773→        ip,
   774→      },
   775→    }).then(() => {
   776→      // Increment user credits
   777→      if (userId) {
   778→        db.user.update({
   779→          where: { id: userId },
   780→          data: { creditsUsed: { increment: 1 } },
   781→        }).catch(() => {});
   782→      }
   783→    }).catch(() => {});
   784→
   785→    // Get remaining credits for response
   786→    let creditsRemaining: number | undefined;
   787→    let userPlan: string | undefined;
   788→    let creditCost: number | undefined;
   789→    if (userId) {
   790→      try {
   791→        const user = await db.user.findUnique({ where: { id: userId }, select: { credits: true, plan: true } });
   792→        if (user) {
   793→          creditsRemaining = user.credits;
   794→          userPlan = user.plan;
   795→        }
   796→      } catch {}
   797→    }
   798→
   799→    return NextResponse.json({
   800→      success: true,
   801→      posts,
   802→      platform,
   803→      topic,
   804→      tone,
   805→      generatedAt: new Date().toISOString(),
   806→      creditsRemaining,
   807→      plan: userPlan,
   808→      creditCost: CREDIT_COST_GENERATE,
   809→    });
   810→  } catch (error) {
   811→    console.error("Generation error:", error);
   812→    const msg = error instanceof Error ? error.message : String(error);
   813→    return NextResponse.json({ error: `Server error: ${msg}` }, { status: 500 });
   814→  }
   815→}
