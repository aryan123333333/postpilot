   262→      {
   263→        role: "assistant",
   264→        content: `You are PostPilot, the world's best social media content strategist. You have 15 years of experience helping brands go viral. You understand each platform's algorithm, culture, and content format at an expert level. Your content never feels AI-generated — it feels like it was written by a top-tier creator who genuinely lives and breathes that platform. You NEVER violate platform-specific format rules.`,
   265→      },
   266→      {
   267→        role: "user",
   268→        content: prompt,
   269→      },
   270→    ],
   271→    thinking: { type: "disabled" },
   272→  });
   273→
   274→  return completion.choices[0]?.message?.content || "Unable to generate content. Please try again.";
   275→}

   359→/*  Post parser — multi-strategy with length validation                */

   361→
   362→function parsePosts(raw: string, platform: string): string[] {
   363→  const profile = PLATFORM_PROFILES[platform] || PLATFORM_PROFILES.twitter;
   364→  let posts: string[] = [];
   365→
   366→  // Pre-clean: strip trailing incomplete delimiter artifacts at the very end
   367→  if (raw.endsWith("POST---")) {
   368→    raw = raw.slice(0, -8).trim();
   369→  }
   370→
   371→  // Strategy 1: Primary delimiter (most reliable)
   372→  if (raw.includes("---POST---")) {
   373→    posts = raw.split("---POST---").map((p) => p.trim()).filter((p) => p.length > 15);
   374→  }
   375→
   376→  // Strategy 2: Numbered list pattern (1. 2. 3. etc.)
   377→  if (posts.length <= 1) {
   378→    const numbered = raw
   379→      .split(/(?:^|\n)\s*\d+[\.\)]\s+/)
   380→      .map((p) => p.trim())
   381→      .filter((p) => p.length > 15);
   382→    if (numbered.length > posts.length) posts = numbered;
   383→  }
   384→
   385→  // Strategy 3: Split by hashtag clusters (works for short AND long platforms)
   386→  if (posts.length <= 1) {
   387→    const hashSplit = raw
   388→      .split(/\n\s*\n(?=#)/)
   389→      .map((p) => p.trim())
   390→      .filter((p) => p.length > 20);
   391→    if (hashSplit.length > posts.length) posts = hashSplit;
   392→  }
   393→
   394→  // Strategy 4: For long platforms, split by repeated "Post #N" labels
   395→  if (posts.length <= 1) {
   396→    const labelSplit = raw
   397→      .split(/(?:(?:^|\n)\s*(?:Post|POST)\s*#\d+[\s:\-]*)/i)
   398→      .map((p) => p.trim())
   399→      .filter((p) => p.length > 50);
   400→    if (labelSplit.length > posts.length) posts = labelSplit;
   401→  }
   402→
   403→  // Smart split: for ANY platform, if a post is way over maxChars,
   404→  // it likely contains multiple posts merged. Split at natural boundaries.
   405→  {
   406→    const expanded: string[] = [];
   407→    for (const post of posts) {
   408→      if (post.length > profile.maxChars * 1.1) {
   409→        let chunks: string[] = [];
   410→
   411→        // Strategy A: Split at double-newline + hashtag (good for IG, YT)
   412→        chunks = post.split(/\n\n(?=#)/).map((c) => c.trim()).filter((c) => c.length > 15);
   413→
   414→        // Strategy B: Split at double-newline + emoji/numbered header
   415→        if (chunks.length <= 1) {
   416→          chunks = post.split(/\n\s*\n(?=[\d#\-*])/).map((c) => c.trim()).filter((c) => c.length > 15);
   417→        }
   418→
   419→        // Strategy C: Split at "Post #N" labels within the merged block
   420→        if (chunks.length <= 1) {
   421→          chunks = post.split(/(?:^|\n)\s*Post\s*#\d+[\s:\-]*/i).map((c) => c.trim()).filter((c) => c.length > 50);
   422→        }
   423→
   424→        // Strategy D: For very long posts, try double newline boundaries
   425→        if (chunks.length <= 1 && post.length > profile.maxChars * 1.5) {
   426→          chunks = post.split(/\n\s*\n/).map((c) => c.trim()).filter((c) => c.length > 100);
   427→        }
   428→
   429→        if (chunks.length > 1) {
   430→          expanded.push(...chunks);
   431→        } else {
   432→          expanded.push(post);
   433→        }
   434→      } else {
   435→        expanded.push(post);
   436→      }
   437→    }
   438→    posts = expanded;
   439→  }
   440→
   441→  // Clean each post
   442→  posts = posts.map((p) =>
   443→    p
   444→      .replace(/^\s*Post\s*\d+[\s:\-]*/i, "")
   445→      .replace(/^\s*[-—]{2,}\s*/gm, "")
   446→      .replace(/^\s*#{1,}\s*Post\s*\d+/i, "")
   447→      .trim()
   448→  );
   449→
   450→  // Filter out posts that are clearly too short or too long for the platform
   451→  posts = posts.filter((p) => {
   452→    if (p.length < 15) return false;
   453→    if (p.length > profile.maxChars * 1.5) return false;
   454→    return true;
   455→  });
   456→
   457→  return posts;
   458→}

   464→const RATE_LIMIT_MAX = 5;
   465→const RATE_LIMIT_WINDOW = 60_000; // 1 minute
   466→
   467→async function isRateLimited(ip: string): Promise<boolean> {
   468→  const since = new Date(Date.now() - RATE_LIMIT_WINDOW);
   469→  const recentCount = await db.rateLimitLog.count({
   470→    where: {
   471→      ip,
   472→      createdAt: { gte: since },
   473→    },
   474→  });
   475→  return recentCount >= RATE_LIMIT_MAX;
   476→}

   479→  await db.rateLimitLog.create({ data: { ip } }).catch(() => {});

   486→// Credit costs per action
   487→const CREDIT_COST_GENERATE = 1;   // 1 credit per generation
   488→const CREDIT_COST_ENHANCE  = 0;   // free — encourage users to write better prompts
   489→
   490→// Plan limits
   491→const PLAN_CREDITS: Record<string, number> = {
   492→  free: 20,
   493→  pro: 500,
   494→  enterprise: 99999, // effectively unlimited
   495→};
   496→
   497→function getCreditsForPlan(plan: string): number {
   498→  return PLAN_CREDITS[plan] || PLAN_CREDITS.free;
   499→}
   500→
   501→interface CreditInfo {
   502→  allowed: boolean;
   503→  remaining: number;
   504→  cost: number;
   505→  plan: string;
   506→  maxCredits: number;
   507→}
   508→
   509→async function checkAndDeductCredits(userId: string, cost: number): Promise<CreditInfo> {
   510→  const user = await db.user.findUnique({ where: { id: userId } });
   511→  if (!user) return { allowed: true, remaining: 20, cost, plan: 'free', maxCredits: 20 };
   512→
   513→  // Enterprise = unlimited
   514→  if (user.plan === 'enterprise') {
   515→    return { allowed: true, remaining: 99999, cost, plan: 'enterprise', maxCredits: 99999 };
   516→  }
   517→
   518→  const maxCredits = getCreditsForPlan(user.plan);
   519→  const remaining = user.credits;
   520→
   521→  if (remaining < cost) {
   522→    return { allowed: false, remaining: 0, cost, plan: user.plan, maxCredits };
   523→  }
   524→
   525→  // Deduct credits
   526→  await db.user.update({
   527→    where: { id: userId },
   528→    data: { credits: { decrement: cost } },
   529→  });
   530→
   531→  return {
   532→    allowed: true,
   533→    remaining: user.credits - cost,
   534→    cost,
   535→    plan: user.plan,
   536→    maxCredits,
   537→  };
   538→}

   540→/* ------------------------------------------------------------------ */
   541→/*  GET credits — endpoint for frontend to check balance               */
   542→/* ------------------------------------------------------------------ */
   543→
   544→export async function GET(request: NextRequest) {
   545→  try {
   546→    const userId = request.nextUrl.searchParams.get('userId');
   547→    if (!userId) {
   548→      return NextResponse.json({ error: 'Missing userId' }, { status: 400 });
   549→    }
   550→    const user = await db.user.findUnique({
   551→      where: { id: userId },
   552→      select: { credits: true, plan: true, createdAt: true },
   553→    });
   554→    if (!user) {
   555→      return NextResponse.json({ credits: 20, plan: 'free', maxCredits: 20 });
   556→    }
   557→    return NextResponse.json({
   558→      credits: user.credits,
   559→      plan: user.plan,
   560→      maxCredits: getCreditsForPlan(user.plan),
   561→    });
   562→  } catch (error) {
   563→    return NextResponse.json({ error: 'Failed to fetch credits' }, { status: 500 });
   564→  }
   565→}

   567→/* ------------------------------------------------------------------ */
   568→/*  API Route                                                          */
   569→/* ------------------------------------------------------------------ */
   570→
   571→export async function POST(request: NextRequest) {
   572→  try {
   573→    const ip = request.headers.get("x-forwarded-for")?.split(",")[0]?.trim() || "unknown";
   574→
   575→    // Rate limiting by IP (database-backed)
   576→    if (await isRateLimited(ip)) {
   577→      await logRateLimit(ip);
   578→      return NextResponse.json(
   579→        { error: "Rate limit reached. Please wait a moment before generating more posts." },
   580→        { status: 429 }
   581→      );
   582→    }
   583→
   584→    const body = await request.json();
   585→    const { topic, platform, tone, count, mode, brandVoice, userId } = body as GenerateRequest;
   586→
   587→    if (!topic || topic.trim().length < 3) {
   588→      return NextResponse.json({ error: "Please provide a topic (at least 3 characters)" }, { status: 400 });
   589→    }
   590→
   591→    // ENHANCE MODE — just return an enhanced prompt, don't generate posts
   592→    if (mode === "enhance") {
   593→      const enhancePrompt = buildEnhancePrompt(topic.trim());
   594→      const enhanced = await generateWithAI(enhancePrompt);
   595→      return NextResponse.json({
   596→        success: true,
   597→        enhanced: enhanced.trim().replace(/^["']|["']$/g, ""),
   598→      });
   599→    }
   600→
   601→    // Credit check for authenticated users
   602→    if (userId) {
   603→      const creditCheck = await checkAndDeductCredits(userId, CREDIT_COST_GENERATE);
   604→      if (!creditCheck.allowed) {
   605→        return NextResponse.json(
   606→          { error: "No credits remaining. Upgrade your plan for more credits.", creditsRemaining: 0, plan: creditCheck.plan },
   607→          { status: 403 }
   608→        );
   609→      }
   610→    }
   611→
   612→    const validPlatforms = Object.keys(PLATFORM_PROFILES);
   613→    if (!platform || !validPlatforms.includes(platform)) {
   614→      return NextResponse.json({ error: `Invalid platform. Choose from: ${validPlatforms.join(", ")}` }, { status: 400 });
   615→    }
   616→
   617→    const validTones = Object.keys(TONE_INSTRUCTIONS);
   618→    if (!tone || !validTones.includes(tone)) {
   619→      return NextResponse.json({ error: `Invalid tone. Choose from: ${validTones.join(", ")}` }, { status: 400 });
   620→    }
   621→
   622→    const postCount = Math.min(Math.max(count || 3, 1), 10);
   623→    const prompt = buildPrompt({ topic, platform, tone, count: postCount, mode: mode || "generate", brandVoice, userId });
   624→
   625→    const rawContent = await generateWithAI(prompt);
   626→    const posts = parsePosts(rawContent, platform);
   627→
   628→    if (posts.length === 0) {
   629→      return NextResponse.json({ error: "Failed to parse generated content. Please try again." }, { status: 500 });
   630→    }
   631→
   632→    // Log generation to database (fire-and-forget)
   633→    db.generation.create({
   634→      data: {
   635→        userId: userId || null,
   636→        topic: topic.trim(),
   637→        platform,
   638→        tone,
   639→        postCount,
   640→        postsGenerated: posts.length,
   641→        mode: mode || "generate",
   642→        postsJson: JSON.stringify(posts),
   643→        ip,
   644→      },
   645→    }).then(() => {
   646→      // Increment user credits
   647→      if (userId) {
   648→        db.user.update({
   649→          where: { id: userId },
   650→          data: { creditsUsed: { increment: 1 } },
   651→        }).catch(() => {});
   652→      }
   653→    }).catch(() => {});
   654→
   655→    // Get remaining credits for response
   656→    let creditsRemaining: number | undefined;
   657→    let userPlan: string | undefined;
   658→    let creditCost: number | undefined;
   659→    if (userId) {
   660→      const user = await db.user.findUnique({ where: { id: userId }, select: { credits: true, plan: true } });
   661→      if (user) {
   662→        creditsRemaining = user.credits;
   663→        userPlan = user.plan;
   664→      }
   665→    }
   666→
   667→    return NextResponse.json({
   668→      success: true,
   669→      posts,
   670→      platform,
   671→      topic,
   672→      tone,
   673→      generatedAt: new Date().toISOString(),
   674→      creditsRemaining,
   675→      plan: userPlan,
   676→      creditCost: CREDIT_COST_GENERATE,
   677→    });
   678→  } catch (error) {
   679→    console.error("Generation error:", error);
   680→    return NextResponse.json({ error: "Internal server error. Please try again later." }, { status: 500 });
   681→  }
   682→}
